    \textbf{How does deadline misses occur?}
    \vspace{5mm}
    \begin{itemize}
        \item Preemption (from interrupts or HP tasks)~\cite{Stankovic:1995, Bernat:2001}
        \item Cache Misses~\cite{Milligan:1996, Wang:2012, Altmeyer:2014, Davis:2013}
        \item CPU overloads~\cite{Baruah:1997, Xu:2015, Ernst:2014}
        \item Security attacks~\cite{hashemi2018comparison, sabaliauskaite2017comparison, Knorn:2019}
    \end{itemize}
